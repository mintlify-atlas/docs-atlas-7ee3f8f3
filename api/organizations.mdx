## Organizations

### List organizations

Retrieve a paginated list of organizations.

```typescript
await avala.organizations.list(options?)
```

<ParamField path="options" type="object">
  Optional parameters for pagination.
  
  <ParamField path="limit" type="number">
    Maximum number of organizations to return per page.
  </ParamField>
  
  <ParamField path="cursor" type="string">
    Cursor for pagination to fetch the next page of results.
  </ParamField>
</ParamField>

<ResponseField name="items" type="Organization[]">
  Array of organization objects.
  
  <ResponseField name="uid" type="string">
    Unique identifier for the organization.
  </ResponseField>
  
  <ResponseField name="name" type="string">
    Name of the organization.
  </ResponseField>
  
  <ResponseField name="slug" type="string">
    URL-friendly slug for the organization.
  </ResponseField>
  
  <ResponseField name="handle" type="string | null">
    Public handle for the organization.
  </ResponseField>
  
  <ResponseField name="description" type="string | null">
    Description of the organization.
  </ResponseField>
  
  <ResponseField name="logo" type="string | null">
    URL to the organization's logo.
  </ResponseField>
  
  <ResponseField name="website" type="string | null">
    Organization's website URL.
  </ResponseField>
  
  <ResponseField name="industry" type="string | null">
    Industry the organization operates in.
  </ResponseField>
  
  <ResponseField name="visibility" type="string | null">
    Visibility setting for the organization.
  </ResponseField>
  
  <ResponseField name="memberCount" type="number | null">
    Number of members in the organization.
  </ResponseField>
  
  <ResponseField name="teamCount" type="number | null">
    Number of teams in the organization.
  </ResponseField>
  
  <ResponseField name="role" type="string | null">
    Current user's role in the organization.
  </ResponseField>
</ResponseField>

<ResponseField name="nextCursor" type="string | null">
  Cursor for the next page of results.
</ResponseField>

<ResponseField name="previousCursor" type="string | null">
  Cursor for the previous page of results.
</ResponseField>

<ResponseField name="hasMore" type="boolean">
  Whether there are more results available.
</ResponseField>
